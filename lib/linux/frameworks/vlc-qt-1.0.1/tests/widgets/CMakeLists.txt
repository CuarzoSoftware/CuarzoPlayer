PROJECT(test-vlcqt-widgets)

###################
# List of sources #
###################
# Define the UI source files
SET(Test_Widgets_UIs
    TestDual.ui
    TestMain.ui
    TestMetaManager.ui
    TestPlayer.ui
    TestRecorder.ui
)

# Define the C++ source files
SET(Test_Widgets_Srcs
    main.cpp
    TestDualInstance.cpp
    TestDualPlayer.cpp
    TestMain.cpp
    TestMetaManager.cpp
    TestPlayer.cpp
    TestRecorder.cpp
)

IF(${LIBVLC_VERSION} EQUAL 0x020200 OR ${LIBVLC_VERSION} GREATER 0x020200)
    SET(Test_Widgets_UIs
        ${Test_Widgets_UIs}

        EqualizerDialog.ui
    )

    SET(Test_Widgets_Srcs
        ${Test_Widgets_Srcs}

        EqualizerDialog.cpp
    )
ENDIF()


#####################
# Compiler settings #
#####################
# This needs to be added in order for #include to work inside .cpp files
INCLUDE_DIRECTORIES(
    ${CMAKE_SOURCE_DIR}/src
    ${CMAKE_SOURCE_DIR}/src/core
    ${CMAKE_SOURCE_DIR}/src/widgets
    ${CMAKE_BINARY_DIR}/src
    ${CMAKE_BINARY_DIR}/tests/test-widgets
)

# This line tells cmake to create the Test program.
ADD_EXECUTABLE(${PROJECT_NAME} ${Test_Widgets_UIC_Srcs} ${Test_Widgets_Srcs})
ADD_DEPENDENCIES(${PROJECT_NAME} ${VLCQT_CORE} ${VLCQT_WIDGETS})

# Link the required libraries
IF(QT_VERSION MATCHES 4)
    TARGET_LINK_LIBRARIES(${PROJECT_NAME} Qt4::QtGui ${VLCQT_CORE_BINARY} ${VLCQT_WIDGETS_BINARY})
ELSE()
    TARGET_LINK_LIBRARIES(${PROJECT_NAME} Qt5::Widgets ${VLCQT_CORE_BINARY} ${VLCQT_WIDGETS_BINARY})
ENDIF()

IF(STATIC)
    TARGET_LINK_LIBRARIES(${PROJECT_NAME} ${LIBVLC_LIBRARY} ${LIBVLCCORE_LIBRARY})
ENDIF()


################
# Installation #
################
INSTALL(TARGETS ${PROJECT_NAME} RUNTIME DESTINATION ${CMAKE_INSTALL_PREFIX}/bin)
